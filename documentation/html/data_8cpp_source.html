348 /* Returns the minimum number of vnfs to be installed for a SFC. */
349 const int Data::getVnfLB(double B, int nbSections) const
350 {
351  /* Get max number of most available nodes per section violating the availability requirement */
352  int vnfs_per_section = 1;
353  while (std::pow(getParallelAvailability(getNMostAvailableNodes(vnfs_per_section)), nbSections) < B){
354  vnfs_per_section++;
355  }
356  int base = std::max(vnfs_per_section - 1, 1);
357  int result = base*nbSections;
358  /* Try to improve the number of vnfs */
359  double section_avail = getParallelAvailability(getNMostAvailableNodes(base));
360  double chain_avail = std::pow(section_avail, nbSections);
361  double new_section_avail = getParallelAvailability(getNMostAvailableNodes(base+1));
362  while (chain_avail < B){
363  chain_avail = (chain_avail*new_section_avail)/section_avail;
364  result++;
365  }
366 
367  return result;
368 }

372 const int Data::getVnfLB(const std::vector<bool> &NODES, const int NB_SECTIONS, const double AVAIL_REQUIRED) const{
