789 void Model::buildApproximationFunctionAvail(int demand, IloNumArray &breakpoints, IloNumArray &slopes){
790  //initialize parameters
791  breakpoints.clear();
792  slopes.clear();
793 
794  // get touch points and breakpoints
795  IloNumArray touchs(env);
796  buildAvailTouchs(demand, touchs);
797  buildAvailBreakpoints(demand, breakpoints, touchs);
798 
799  // get slopes
800  switch (data.getInput().getApproximationType()){
801  case Input::APPROXIMATION_TYPE_RESTRICTION:
802  slopes.add( 1.0/touchs[0] );
803  for (int i = 0; i < touchs.getSize()-1; i++){
804  double delta_X = touchs[i+1] - touchs[i];
805  double delta_Y = std::log(touchs[i+1]) - std::log(touchs[i]);
806  slopes.add( delta_Y/delta_X );
807  }
808  slopes.add( 0 );
809 
810  break;
811  case Input::APPROXIMATION_TYPE_RELAXATION:
812  for (unsigned int i = 0; i < touchs.getSize(); i++){
813  double tangent = 1.0/touchs[i];
814  slopes.add(tangent);
815  }
816  break;
817  default:
818  throw IloCplex::Exception(-1, "ERROR: Unexpected availability approx !");
819  }
